Приклад схеми URL

HTML-код

  Приклад схеми URL
</a>

Обробник подій

В якості прикладу використання обробника подій наведено кнопку яка викликає спливаюче вікно з повідомленням.

HTML-код 
<button onclick="alert('Ви натиснули на кнопку')">Кнопка</button>

Підстановка(entity)

В якості прикладу використання підстановки наведено скрипт який підставляє поточний час в текстове поле.

HTML-код 
<input type="text" id="time-field">

<script>
  function setTime(){
    date = new Date();
    hours = date.getHours();
    minutes = date.getMinutes();
    seconds = date.getSeconds();
    document.getElementById("time-field").value = "Поточний час - "+
    hours + ":" + minutes + "." + seconds; 
  }
  setInterval(setTime, 1000);
</script>

var arrA = [5, 2, 1, 1, 2, 1, 1, 5, 1, 1];
var arrB = [2, 1, 3, 1, 3, 1, 2, 1, 1, 4];
var arrC = [];

function calculateC() {
  arrC = calcC();
  document.getElementById("arrayC1").innerHTML = 
  arrC.map(round).join(", ");

  arrC = changeElements(arrC);
  document.getElementById("arrayC2").innerHTML = 
  arrC.map(round).join(", ");

  arrC = bubbleSort(arrC);
  document.getElementById("arrayC3").innerHTML = 
  arrC.map(round).join(", ");
}

function calcC() {
  var c = []
  for (var i = 0; i < 10; i++) {
      if (arrA[i] == arrB[i]) {
        c[i] = 1;
      } else {
        c[i] = (1 / (arrA[i] - arrB[i]));
      }
  }
  return c
}

function changeElements(arr) {
   var arr_first = arr[0]
  arr[0] = arr[9];
  arr[9] = arr_first;
  return arr
}

function bubbleSort(arr){
  for(var i = 0; i < arr.length; i++){
    for(var j = 0; j < ( arr.length - i -1 ); j++){
      if(arr[j] > arr[j+1]){
        var temp = arr[j]
        arr[j] = arr[j + 1]
        arr[j+1] = temp
      }
    }
  }
  return arr
}


function round(element){
  return element.toFixed(2);
}
