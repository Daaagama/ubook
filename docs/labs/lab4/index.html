Схема URL

В якості прикладу використання схеми URL наведено посилання яке викликає спливаюче вікно.

Приклад схеми URL

HTML-код 

<a href="JavaScript:alert('Ви натиснули на посилання');">
  Приклад схеми URL
</a>

Обробник подій

В якості прикладу використання обробника подій наведено кнопку яка викликає спливаюче вікно з повідомленням.

HTML-код 
<button onclick="alert('Ви натиснули на кнопку')">Кнопка</button>

Підстановка(entity)

В якості прикладу використання підстановки наведено скрипт який підставляє поточний час в текстове поле.

HTML-код 
<input type="text" id="time-field">

<script>
  function setTime(){
    date = new Date();
    hours = date.getHours();
    minutes = date.getMinutes();
    seconds = date.getSeconds();
    document.getElementById("time-field").value = "Поточний час - "+
    hours + ":" + minutes + "." + seconds; 
  }
  setInterval(setTime, 1000);
</script>

function generateRandomArray() {
    return Array.from({length: 20}, () => Math.floor(Math.random() * 40)-20);
}

function getMinPositiveElement(array) {
    result = Math.max.apply(null, array);
    for (var i = 0; i < array.length; i++) {
        if ((array[i] > 0) && (result > array[i])) {
            result = array[i];
        }
    }
    return result;
}

function getMaxNegativeElement(array) {
    result = Math.min.apply(null, array);
    for (var i = 0; i < array.length; i++) {
        if ((array[i] < 0) && (result < array[i])) {
            result = array[i];
        }
    }
    return result;
}

function selectionSort(inputArr) { 
    let n = inputArr.length;
        
    for(let i = 0; i < n; i++) {
        let min = i;
        for(let j = i; j < n; j++) {
            if(inputArr[j] < inputArr[min]) {
                min=j; 
            }
         }
         if (min != i) {
             let tmp = inputArr[i]; 
             inputArr[i] = inputArr[min];
             inputArr[min] = tmp;      
        }
    }
    return inputArr.reverse();
}

function task4v7() {
    var a = generateRandomArray();
    var ra = [...a];
    var r1 = getMaxNegativeElement(a);
    var r2 = getMinPositiveElement(a);
    var r3 = selectionSort(a);
    document.getElementById("task-4-variant-7").innerHTML = ra + "<br>"+r1+"<br>"+r2+"<br>"+r3;
}
